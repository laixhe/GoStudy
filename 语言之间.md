##### 1. Golang
```
package main
import "fmt"

func main() {
	lists := []string{"a", "b", "c"}
	maps := map[string]int{"a": 1, "b": 2, "c": 3}
	fmt.Println(lists)
	fmt.Println(maps)
}
```
